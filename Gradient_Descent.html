def gradientDescent(x, y):
    bCurrent = mCurrent = 0
    iterations = 1500
    learninRate = 0.012
    n = len(x)
    plt.scatter(x,y, marker="x")
    for i in range(iterations):
        y_predicted = bCurrent * x + bCurrent
        mDerivative = -(2 / n) * sum(x*(y - y_predicted))
        bDerivative = -(2 / n) * sum((y - y_predicted))
        mCurrent = mCurrent - learninRate * mDerivative
        bCurrent = bCurrent - learninRate * bDerivative
        MSE = (1 / n) * sum((y - y_predicted) ** 2)
        print("m: {},b: {}, iteration: {}, mse: {}".format(mCurrent,bCurrent,i,MSE))
        
        %matplotlib inline
        print(plt.plot(x,y_predicted))
